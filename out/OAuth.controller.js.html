exports.signinGoogle = (req, res) => {
   
  User.findOne({
    googleId: req.body.googleId
  })
    .exec((err, user) => {
      if (err) {
        res.status(500).send({ message: err });
        return;
      }
      if (!user) {
        return res.status(404).send({ message: "User Not found." });
      }
      
      var token = jwt.sign({ id: user.id }, config.secret, {
